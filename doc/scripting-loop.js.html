const events_1 = require("events");
class LoopEvents {
}
LoopEvents.OnLoopResume = 'AVGEvent_LoopResume';
LoopEvents.OnLoopData = 'AVGEvent_LoopData';
LoopEvents.OnLoopEnd = 'AVGEvent_LoopEnd';
exports.LoopEvents = LoopEvents;
;
class AVGScriptingLoop {
    constructor() {
        this._scripts = [];
        this._cursor = 0;
        this._handle = null;
        this._loopEvent = new events_1.EventEmitter();
    }
    /**
     * Add an story to the scripting loop system
     *
     * @param {AVGStory} story
     * @memberof AVGScriptingLoop
     */
    addStory(story) {
        let scripts = story.getScripts();
        this._scripts = this._scripts.concat(scripts);
    }
    run() {
        if (this._scripts.length === 0) {
            this._loopEvent.emit(LoopEvents.OnLoopEnd);
            return this._loopEvent;
        }
        this._handle = setInterval(() => {
            let current = this._scripts[this._cursor];
            if (!current) {
                this._loopEvent.emit(LoopEvents.OnLoopEnd);
                this.reset();
                return;
            }
            this._loopEvent.emit(LoopEvents.OnLoopData, current);
            current.execute && current.execute().then((data) => {
                this._loopEvent.emit(LoopEvents.OnLoopResume);
            }, _ => { });
            this.waitFor();
        }, 1);
        return this._loopEvent;
    }
    waitFor() {
        console.log('Scripting loop was waiting for ...');
        this.reset();
        this._loopEvent.once(LoopEvents.OnLoopResume, () => {
            this._cursor++;
            this.run();
        });
    }
    reset() {
        if (this._handle) {
            clearInterval(this._handle);
        }
    }
}
exports.AVGScriptingLoop = AVGScriptingLoop;
